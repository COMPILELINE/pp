11. Write a java application to create server and client program using TCP/IP protocol.

//Server side
import java.net.*;
import java.io.*;
class Main
{
    public static void main(String args[])
    {
        try
        {
            ServerSocket s1=new ServerSocket(420);
            Socket s2=s1.accept();
            InputStream in=s2.getInputStream();
            int ch;
            while((ch=in.read())!=-1)
            {
                System.out.print((char)ch);
            }
            in.close();
            s2.close();
            s1.close();
        }
        catch(Exception e)
        {
            System.out.println(e);
        }
    }
}

class Main
{

    {

        {
            DatagramSocket ds=new DatagramSocket(101);
            String str="good morning";
            byte b[]=str.getBytes();

            ds.send(dp);
            ds.close();
        }
        catch(Exception e)
        {
            System.out.println(e);
        }
    }
}
